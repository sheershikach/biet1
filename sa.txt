//SWAPPING OF 2 NUMBERS USING 3RD VARIABLE.
#include<stdio.h>
int main()
{
int a=10, b=20, c;
printf("Before swapping A=%d B=%d", a,b);
c=a;
a=b;
b=c;
printf("After swapping A=%d B=%d",a,b);
}
//SWAPPING OF 2 NUMBERS WITHOUT USING THIRD VARIABLE
#include<stdio.h>
int main()
{
int a=10,b=20;
printf("Before swapping A=%d B=%d", a,b);
a=a+b;
b=a-b;
a=a-b;
printf("After swapping A=%d B=%d",a,b);
}
